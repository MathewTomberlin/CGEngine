682 ScriptEvent MouseOverlapReleaseEvent = [](ScArgs args) {
683 //Get the mouse release event from the script input data
684 MouseReleaseInput* evt = args.script->getInput().getDataPtr<MouseReleaseInput>("evt");
685 if (evt == nullptr) return;
686
687 //If the caller's bounds contains the mouse position (converted from View Space)

744 //Get the key released event from the script input data
745 KeyReleaseInput* evt = args.script->getInput().getDataPtr<KeyReleaseInput>("evt");
746 if(evt == nullptr) return;

752 optional<id_t> updateId = (*iterator).second;

754 //Erase the key hold update script
