23 ScriptEvent tileClickEvt = [this](ScArgs args) {
24 MouseReleaseInput* mouseEvt = args.script->getInputDataPtr<MouseReleaseInput>("evt");
25 Tilemap* tilemap = args.caller->get<Tilemap*>();
26 if (mouseEvt == nullptr || tilemap == nullptr) return;
27
28 //Convert the mouse click position to global space, then to tilemap local space, then as a flattened id
29 clickedTilemapPos = args.caller->viewToLocal(mouseEvt->position);
30 int tileId = tilemap->localPosToTileID(clickedTilemapPos.value());
31 //Set assignedType on RMB or assign the tile type on LMB
32 if (mouseEvt->button == Mouse::Button::Right) {
33 assignedType = tilemap->query(tileId);
34 } else {
35 tilemap->assign(tileId, assignedType);
36 }
37 };
